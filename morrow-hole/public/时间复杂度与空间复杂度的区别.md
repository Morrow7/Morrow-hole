for (let i = 0; i < n; i++) {} // O(n)

for (let i = 0; i < n; i++) {
  for (let j = 0; j < n; j++) {}
} // O(n²)

***

空间复杂度本质是：

- 除了原始数据，还额外用了多少内存

![image-20260127234044361](C:\Users\45379\AppData\Roaming\Typora\typora-user-images\image-20260127234044361.png)

// O(1) 空间
function sum(arr) {
  let total = 0;
  for (let x of arr) total += x;
}

// O(n) 空间
function copy(arr) {
  return [...arr];
}
